{-# OPTIONS --cubical #-}

module Leibniz.Base where

open import Prelude
open import Extra.Algebra

import Cubical.Data.Nat.Base as N


infix 20 _1b _2b

-- The type of the 1-2 binary numbers
data Leibniz : Set where
  0b : Leibniz
  _1b : Leibniz → Leibniz
  _2b : Leibniz → Leibniz

-- The successor operation on binary numbers
suc : Leibniz → Leibniz
suc 0b     = 0b 1b
suc (n 1b) = n 2b
suc (n 2b) = (suc n) 1b

-- Addition on binary numbers
_+_ : Leibniz → Leibniz → Leibniz
0b     + y  = y
x      + 0b = x
(x 1b) + (y 1b) = (x + y) 2b
(x 1b) + (y 2b) = suc (x + y) 1b
(x 2b) + (y 1b) = suc (x + y) 1b
(x 2b) + (y 2b) = suc (x + y) 2b

-- Right shift the binary number by a peano number of steps
_>>_ : N.ℕ → Leibniz → Leibniz
ℕ.zero  >> y      = y
ℕ.suc x >> 0b     = 0b
ℕ.suc x >> (y 1b) = x >> y
ℕ.suc x >> (y 2b) = x >> y

infix 6 _<_ _≤_

-- Inequality on binary numbers
data _<_ : Leibniz → Leibniz → Type
_≤_ : Leibniz → Leibniz → Type

data _<_ where
  0b<1b :         0b   < y 1b
  0b<2b :         0b   < y 2b
  1b<1b : x < y → x 1b < y 1b
  1b<2b : x ≤ y → x 1b < y 2b
  2b<1b : x < y → x 2b < y 1b
  2b<2b : x < y → x 2b < y 2b

x ≤ y = (x < y) ⊎ (x ≡ y)

-- From peano to leibniz
fromℕ : ℕ → Leibniz
fromℕ ℕ.zero    = 0b
fromℕ (ℕ.suc n) = suc (fromℕ n)

-- From leibniz to peano
⟦_⟧ : Leibniz → ℕ
⟦ 0b ⟧   = 0
⟦ n 1b ⟧ = 1 N.+ 2 N.· ⟦ n ⟧ 
⟦ n 2b ⟧ = 2 N.+ 2 N.· ⟦ n ⟧

-- Structures
MagmaL : Magma
fst MagmaL = Leibniz
snd MagmaL = _+_
